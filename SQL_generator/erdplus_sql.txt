CREATE TABLE SEXO
(
  ID INT NOT NULL,
  NOMBRE VARCHAR(10) NOT NULL,
  PRIMARY KEY (ID)
);

CREATE TABLE FORMSUNARP
(
  ID INT NOT NULL,
  NRO_PROPIEDADES INT NOT NULL,
  NRO_VEHICULOS INT NOT NULL,
  OBSERVACIONES VARCHAR(200) NOT NULL,
  PRIMARY KEY (ID)
);

CREATE TABLE FORMMINTRABAJO
(
  ID INT NOT NULL,
  NOMBRE_LUGAR_TRABAJO VARCHAR(30) NOT NULL,
  RUC_LUGAR_TRABAJO VARCHAR(11) NOT NULL,
  MONTO_INGRESO INT NOT NULL,
  OBSERVACIONES VARCHAR(200) NOT NULL,
  PRIMARY KEY (ID)
);

CREATE TABLE SECRETARIOJUDICIAL
(
  ID INT NOT NULL,
  DNI VARCHAR(8) NOT NULL,
  NOMBRES VARCHAR(45) NOT NULL,
  APELLIDOS VARCHAR(45) NOT NULL,
  NRO_COLEGIATURA INT NOT NULL,
  SEXO_ID INT NOT NULL,
  PRIMARY KEY (ID),
  FOREIGN KEY (SEXO_ID) REFERENCES SEXO(ID)
);

CREATE TABLE JUZGADOPAZLETRADO
(
  ID INT NOT NULL,
  DIRECCION VARCHAR(100) NOT NULL,
  PRIMARY KEY (ID)
);

CREATE TABLE ESTADOCIVIL
(
  ID INT NOT NULL,
  NOMBRE VARCHAR(10) NOT NULL,
  PRIMARY KEY (ID)
);

CREATE TABLE GRADOINSTRUCCION
(
  ID INT NOT NULL,
  NOMBRE VARCHAR(30) NOT NULL,
  PRIMARY KEY (ID)
);

CREATE TABLE TIPODEMANDA
(
  ID INT NOT NULL,
  NOMBRE VARCHAR(30) NOT NULL,
  PRIMARY KEY (ID)
);

CREATE TABLE ESTADODEMANDA
(
  ID INT NOT NULL,
  NOMBRE VARCHAR(30) NOT NULL,
  PRIMARY KEY (ID)
);

CREATE TABLE ABOGADO
(
  ID INT NOT NULL,
  DNI VARCHAR(8) NOT NULL,
  NOMBRES VARCHAR(45) NOT NULL,
  APELLIDOS VARCHAR(45) NOT NULL,
  NRO_COLEGIATURA INT NOT NULL,
  SEXO_ID INT NOT NULL,
  PRIMARY KEY (ID),
  FOREIGN KEY (SEXO_ID) REFERENCES SEXO(ID)
);

CREATE TABLE JUEZ
(
  ID INT NOT NULL,
  DNI VARCHAR(8) NOT NULL,
  NOMBRES VARCHAR(45) NOT NULL,
  APELLIDOS VARCHAR(45) NOT NULL,
  NRO_COLEGIATURA INT NOT NULL,
  USUARIO VARCHAR(20) NOT NULL,
  CONTRASENA VARCHAR(20) NOT NULL,
  JUZGADO_ID INT NOT NULL,
  SEXO_ID INT NOT NULL,
  PRIMARY KEY (ID),
  FOREIGN KEY (JUZGADO_ID) REFERENCES JUZGADOPAZLETRADO(ID),
  FOREIGN KEY (SEXO_ID) REFERENCES SEXO(ID),
  UNIQUE (USUARIO)
);

CREATE TABLE FORMRENIEC
(
  ID INT NOT NULL,
  DNI VARCHAR(8) NOT NULL,
  NOMBRES VARCHAR(45) NOT NULL,
  APELLIDOS VARCHAR(45) NOT NULL,
  DIRECCION VARCHAR(100) NOT NULL,
  ESTADOCIVIL_ID INT NOT NULL,
  GRADOINSTRUCCION_ID INT NOT NULL,
  PRIMARY KEY (ID),
  FOREIGN KEY (ESTADOCIVIL_ID) REFERENCES ESTADOCIVIL(ID),
  FOREIGN KEY (GRADOINSTRUCCION_ID) REFERENCES GRADOINSTRUCCION(ID)
);

CREATE TABLE EXPEDIENTE
(
  ID INT NOT NULL,
  NRO_EXPEDIENTE INT NOT NULL,
  OBSERVACIONES VARCHAR(200) NOT NULL,
  FORMRENIEC_ID INT NOT NULL,
  FORMSUNARP_ID INT NOT NULL,
  FORMMINTRABAJO_ID INT NOT NULL,
  SECRETARIOJUDICIAL_ID INT NOT NULL,
  PRIMARY KEY (ID),
  FOREIGN KEY (FORMRENIEC_ID) REFERENCES FORMRENIEC(ID),
  FOREIGN KEY (FORMSUNARP_ID) REFERENCES FORMSUNARP(ID),
  FOREIGN KEY (FORMMINTRABAJO_ID) REFERENCES FORMMINTRABAJO(ID),
  FOREIGN KEY (SECRETARIOJUDICIAL_ID) REFERENCES SECRETARIOJUDICIAL(ID)
);

CREATE TABLE DEMANDADO
(
  ID INT NOT NULL,
  DNI VARCHAR(8) NOT NULL,
  NOMBRES VARCHAR(45) NOT NULL,
  APELLIDOS VARCHAR(45) NOT NULL,
  DIRECCION VARCHAR(100) NOT NULL,
  EXPEDIENTE_ID INT NOT NULL,
  SEXO_ID INT NOT NULL,
  PRIMARY KEY (ID),
  FOREIGN KEY (EXPEDIENTE_ID) REFERENCES EXPEDIENTE(ID),
  FOREIGN KEY (SEXO_ID) REFERENCES SEXO(ID)
);

CREATE TABLE DEMANDANTE
(
  ID INT NOT NULL,
  DNI VARCHAR(8) NOT NULL,
  NOMBRES VARCHAR(45) NOT NULL,
  APELLIDOS VARCHAR(45) NOT NULL,
  DIRECCION VARCHAR(100) NOT NULL,
  EXPEDIENTE_ID INT NOT NULL,
  SEXO_ID INT NOT NULL,
  PRIMARY KEY (ID),
  FOREIGN KEY (EXPEDIENTE_ID) REFERENCES EXPEDIENTE(ID),
  FOREIGN KEY (SEXO_ID) REFERENCES SEXO(ID)
);

CREATE TABLE DEMANDA
(
  ID INT NOT NULL,
  DESCRIPCION VARCHAR(200) NOT NULL,
  NRO_SOLICITUD INT NOT NULL,
  FECHA_EMISION DATE NOT NULL,
  JUEZ_ID INT NOT NULL,
  DEMANDANTE_ID INT NOT NULL,
  DEMANDADO_ID INT NOT NULL,
  ABOGADO_ID INT NOT NULL,
  TIPODEMANDA_ID INT NOT NULL,
  ESTADODEMANDA_ID INT NOT NULL,
  PRIMARY KEY (ID),
  FOREIGN KEY (JUEZ_ID) REFERENCES JUEZ(ID),
  FOREIGN KEY (DEMANDANTE_ID) REFERENCES DEMANDANTE(ID),
  FOREIGN KEY (DEMANDADO_ID) REFERENCES DEMANDADO(ID),
  FOREIGN KEY (ABOGADO_ID) REFERENCES ABOGADO(ID),
  FOREIGN KEY (TIPODEMANDA_ID) REFERENCES TIPODEMANDA(ID),
  FOREIGN KEY (ESTADODEMANDA_ID) REFERENCES ESTADODEMANDA(ID)
);
